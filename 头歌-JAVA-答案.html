2. for循环的进阶使用-嵌套循环（2）

package step2;

public class ForPractice2 {
	public static void main(String[] args) {
		/*****start*****/
		
		//在这里打印出正三角形 
		for(int m=1;m<=10;m++){
            for(int n=1;n<=m;n++){
                System.out.print("*");
            }
            System.out.println();
        }

		System.out.println("——————————我是华丽的分界线——————————");
		//在这里打印出倒三角形
		for(int x=1;x<=10;x++){
            for(int y=10;y>=x;y--){
                System.out.print("*");
            }
            System.out.println();
        }

		/*****end*****/
	}
}
